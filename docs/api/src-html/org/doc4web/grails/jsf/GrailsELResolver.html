001    package org.doc4web.grails.jsf;
002    
003    import org.codehaus.groovy.grails.commons.GrailsApplication;
004    import org.springframework.beans.factory.BeanFactory;
005    import org.springframework.context.MessageSource;
006    import org.springframework.web.jsf.el.SpringBeanFacesELResolver;
007    
008    import javax.el.ELContext;
009    import javax.el.ELException;
010    import javax.el.PropertyNotWritableException;
011    import javax.faces.context.FacesContext;
012    import java.util.Locale;
013    
014    /**
015     * org.doc4web.grails.jsf.facelets.org.doc4web.grails.jsf
016     * <p/>
017     * Date: 15 mars 2010
018     * Time: 18:58:12
019     *
020     * @author pred
021     */
022    public class GrailsELResolver extends SpringBeanFacesELResolver {
023    
024        private static final Object[] NO_ARGS = new Object[]{};
025        private static final String MESSAGESOURCE_SHORTCUT = "m";
026        private static final String APPLICATION_SHORTCUT = "app";
027    
028        @Override
029        public Object getValue(ELContext elContext, Object base, Object property) throws ELException {
030            if (base == null) {
031                String beanName = property.toString();
032    
033                if (beanName.equals(MESSAGESOURCE_SHORTCUT))
034                    beanName = "messageSource";
035                else if (beanName.equals(APPLICATION_SHORTCUT)) {
036                    beanName = GrailsApplication.APPLICATION_ID;
037                }
038    
039                BeanFactory bf = getBeanFactory(elContext);
040                if (bf.containsBean(beanName)) {
041                    if (logger.isTraceEnabled()) {
042                        logger.trace("Successfully resolved variable '" + beanName + "' in Spring BeanFactory");
043                    }
044                    elContext.setPropertyResolved(true);
045                    return bf.getBean(beanName);
046                }
047            } else if (base instanceof MessageSource) {
048                MessageSource messageSource = (MessageSource) base;
049                Locale locale = FacesContext.getCurrentInstance().getViewRoot()
050                        .getLocale();
051                elContext.setPropertyResolved(true);
052                try {
053                    return messageSource.getMessage((String) property, NO_ARGS, locale);
054                } catch (org.springframework.context.NoSuchMessageException m) {
055                    return (String) property;
056                }
057            }
058            return null;
059        }
060    
061        @Override
062        public Class<?> getType(ELContext elContext, Object base, Object property) throws ELException {
063            if (base == null) {
064                String beanName = property.toString();
065    
066                if (beanName.equals(MESSAGESOURCE_SHORTCUT))
067                    beanName = "messageSource";
068    
069                BeanFactory bf = getBeanFactory(elContext);
070                if (bf.containsBean(beanName)) {
071                    elContext.setPropertyResolved(true);
072                    return bf.getType(beanName);
073                }
074            } else if (base instanceof MessageSource) {
075                return String.class;
076            }
077            return null;
078        }
079    
080        @Override
081        public void setValue(ELContext elContext, Object base, Object property, Object value) throws ELException {
082            if (base == null) {
083                String beanName = property.toString();
084    
085                if (beanName.equals(MESSAGESOURCE_SHORTCUT))
086                    beanName = "messageSource";
087    
088                BeanFactory bf = getBeanFactory(elContext);
089                if (bf.containsBean(beanName)) {
090                    throw new PropertyNotWritableException(
091                            "Variable '" + beanName + "' refers to a Spring bean which by definition is not writable");
092                }
093            } else if (base instanceof MessageSource && property != null) {
094                throw new PropertyNotWritableException(
095                        "MessageSource: '" + property + "' refers to a Spring bean which by definition is not writable");
096            }
097        }
098    
099        @Override
100        public boolean isReadOnly(ELContext elContext, Object base, Object property) throws ELException {
101            if (base == null) {
102                String beanName = property.toString();
103    
104                if (beanName.equals(MESSAGESOURCE_SHORTCUT))
105                    beanName = "messageSource";
106    
107                BeanFactory bf = getBeanFactory(elContext);
108                if (bf.containsBean(beanName)) {
109                    return true;
110                }
111            } else if (base instanceof MessageSource) {
112                return true;
113            }
114            return false;
115        }
116    }
