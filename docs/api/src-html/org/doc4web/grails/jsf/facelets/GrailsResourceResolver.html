027    public class GrailsResourceResolver extends ResourceResolver {
028    
029        private GroovyPageResourceLoader resourceLoader;
030        private ServletContextResourceLoader servletContextLoader;
031    
032    
033        public GrailsResourceResolver() {
034            this.resourceLoader = (GroovyPageResourceLoader) ApplicationHolder.getApplication().getMainContext().getBean(GroovyPageResourceLoader.BEAN_ID);
035        }
036    
037        public URL resolveUrl(String s) {
038            if (servletContextLoader == null) {
039                ServletContext sc = (ServletContext) FacesContext.getCurrentInstance().getExternalContext().getContext();
040                this.servletContextLoader = new ServletContextResourceLoader(sc);
041            }
042            try {
043                return this.getResourceForUri(s).getURL();
044                /*return new URL("file", "",
045                        "PATH_TO_FACELETS_FILES_GOES_HERE" + s);*/
046            } catch (MalformedURLException e) {
047                e.printStackTrace();
048                return null;
049            } catch (IOException e) {
050                e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
051                return null;
052            }
053        }
054    
055        private Resource getResourceForUri(String uri) {
056            Resource r;
057            r = getResourceWithinContext(uri);
058            if (r == null || !r.exists()) {
059                // try plugin
060                String pluginUri = GrailsResourceUtils.WEB_INF + uri;
061                r = getResourceWithinContext(pluginUri);
062                if (r == null || !r.exists()) {
063                    uri = getUriWithinGrailsViews(uri);
064                    return getResourceWithinContext(uri);
065                }
066            }
067            return r;
068        }
069    
070        protected String getUriWithinGrailsViews(String relativeUri) {
071            StringBuilder buf = new StringBuilder();
072            String[] tokens;
073            if (relativeUri.startsWith("/"))
074                relativeUri = relativeUri.substring(1);
075    
076    
077            if (relativeUri.indexOf('/') > -1)
078                tokens = relativeUri.split("/");
079            else
080                tokens = new String[]{relativeUri};
081    
082            buf.append(GrailsApplicationAttributes.PATH_TO_VIEWS);
083            for (String token : tokens) {
084                buf.append('/').append(token);
085            }
086    
087            return buf.toString();
088        }
089    
090        private Resource getResourceWithinContext(String uri) {
091            if (resourceLoader == null)
092                throw new IllegalStateException("ResourceResolver not initialised correctly, no [resourceLoader] specified!");
093            if (Environment.getCurrent().isReloadEnabled() && Metadata.getCurrent().isWarDeployed()) {
094                return resourceLoader.getResource(uri);
095            }
096            Resource r = servletContextLoader.getResource(uri);
097            if (r.exists()) return r;
098            return resourceLoader.getResource(uri);
099        }
100    }
