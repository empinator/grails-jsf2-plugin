030    public class RedirectDynamicMethod extends org.codehaus.groovy.grails.web.metaclass.RedirectDynamicMethod {
031    
032        public static final String BEAN_ARG = "bean";
033        public static final String VIEW_ARG = "view";
034        public static final String ACTION_ARG = "action";
035        public static final String DEFAULT_VIEW = "index";
036    
037        private String extension = (String) ConfigurationHolder.getFlatConfig().get("icefaces.extension");
038    
039        public RedirectDynamicMethod(ApplicationContext applicationContext) {
040            super(applicationContext);
041        }
042    
043        @Override
044        public Object invoke(Object target, String methodName, Object[] arguments) {
045            if (arguments.length == 0)
046                throw new MissingMethodException(METHOD_SIGNATURE, target.getClass(), arguments);
047    
048            Map argMap = arguments[0] instanceof Map ? (Map) arguments[0] : Collections.EMPTY_MAP;
049            if (argMap.size() == 0) {
050                throw new MissingMethodException(METHOD_SIGNATURE, target.getClass(), arguments);
051            }
052    
053            GrailsWebRequest webRequest = (GrailsWebRequest) RequestContextHolder.currentRequestAttributes();
054    
055            HttpServletRequest request = webRequest.getCurrentRequest();
056            HttpServletResponse response = webRequest.getCurrentResponse();
057    
058            if (request.getAttribute(GRAILS_REDIRECT_ISSUED) != null) {
059                throw new CannotRedirectException("Cannot issue a redirect(..) here. A previous call to redirect(..) has already redirected the response.");
060            }
061            if (response.isCommitted()) {
062                throw new CannotRedirectException("Cannot issue a redirect(..) here. The response has already been committed either by another redirect or by directly writing to the response.");
063            }
064    
065            Object uri = argMap.get(ARGUMENT_URI);
066            String url = argMap.containsKey(ARGUMENT_URL) ? argMap.get(ARGUMENT_URL).toString() : null;
067            String bean = (String) argMap.get(BEAN_ARG);
068            String action = (String) argMap.get(ACTION_ARG);
069            String view = (String) argMap.get(VIEW_ARG);
070            String sourceBean = GrailsNameUtils.getLogicalPropertyName(target.getClass().getName(), BeanArtefactHandler.TYPE);
071    
072            if (action == null) {
073                if (bean != null) {
074                    action = "/" + bean + "/" + (view == null ? DEFAULT_VIEW : view );
075                }else{
076                    action = view;
077                }
078            }
079            GrailsApplicationAttributes attrs = webRequest.getAttributes();
080    
081            FacesContext fc = FacesContext.getCurrentInstance();
082            if (action != null) {
083                //fc.getApplication().getNavigationHandler().handleNavigation(fc, fc.getViewRoot().getViewId(), action);
084                ViewHandler viewHandler = Util.getViewHandler(fc);
085                ExternalContext extContext = fc.getExternalContext();
086                String newPath = viewHandler.getActionURL(fc, action);
087                try {
088                    extContext.redirect(extContext.encodeActionURL(newPath));
089                    fc.responseComplete();
090                } catch (IOException e) {
091                    throw new CannotRedirectException("IO Exception " + e.getMessage());
092                }
093                return null;
094            }
095            try {
096                String actualUri = null;
097                if (uri != null) {
098                    actualUri = attrs.getApplicationUri(request) + uri.toString();
099                } else if (url != null) {
100                    actualUri = url;
101                }
102                fc.getExternalContext().redirect(actualUri);
103            } catch (IOException e) {
104                throw new CannotRedirectException("IO Exception " + e.getMessage());
105            }
106            return null;
107        }
108    }
