025    public class RenderDynamicMethod extends org.codehaus.groovy.grails.web.metaclass.RedirectDynamicMethod {
026    
027        private static final String METHOD_SIGNATURE = "render";
028    
029        private String extension = (String) ConfigurationHolder.getFlatConfig().get("jsf.extension");
030    
031        public RenderDynamicMethod(ApplicationContext applicationContext) {
032            super(applicationContext);
033        }
034    
035        @Override
036        public Object invoke(Object target, String methodName, Object[] arguments) {
037            if (arguments.length == 0)
038                throw new MissingMethodException(METHOD_SIGNATURE, target.getClass(), arguments);
039    
040            Map argMap = arguments[0] instanceof Map ? (Map) arguments[0] : Collections.EMPTY_MAP;
041            if (argMap.size() == 0) {
042                throw new MissingMethodException(METHOD_SIGNATURE, target.getClass(), arguments);
043            }
044    
045            GrailsWebRequest webRequest = (GrailsWebRequest) RequestContextHolder.currentRequestAttributes();
046    
047            HttpServletRequest request = webRequest.getCurrentRequest();
048            HttpServletResponse response = webRequest.getCurrentResponse();
049    
050            if (request.getAttribute(GRAILS_REDIRECT_ISSUED) != null) {
051                throw new CannotRedirectException("Cannot issue a redirect(..) here. A previous call to redirect(..) has already redirected the response.");
052            }
053            if (response.isCommitted()) {
054                throw new CannotRedirectException("Cannot issue a redirect(..) here. The response has already been committed either by another redirect or by directly writing to the response.");
055            }
056    
057            String bean = (String) argMap.get(RedirectDynamicMethod.BEAN_ARG);
058            String action = (String) argMap.get(RedirectDynamicMethod.ACTION_ARG);
059            String view = (String) argMap.get(RedirectDynamicMethod.VIEW_ARG);
060            String sourceBean = GrailsNameUtils.getLogicalPropertyName(target.getClass().getName(), BeanArtefactHandler.TYPE);
061    
062            if (action == null) {
063                if (bean == null && view != null) {
064                    bean = sourceBean;
065                } else if (bean != null && view == null) {
066                    view = RedirectDynamicMethod.DEFAULT_VIEW;
067                }
068                if (bean != null) {
069                    action = "/" + bean + "/" + view;
070                }
071            }
072            FacesContext fc = FacesContext.getCurrentInstance();
073            if (action != null) {
074                fc.getApplication().getNavigationHandler().handleNavigation(fc, fc.getViewRoot().getViewId(),
075                        action);
076                return null;
077            }
078    
079            return null;
080        }
081    }
