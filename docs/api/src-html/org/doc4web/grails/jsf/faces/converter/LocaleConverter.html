001    package org.doc4web.grails.jsf.faces.converter;
002    
003    import javax.faces.component.UIComponent;
004    import javax.faces.context.FacesContext;
005    import javax.faces.convert.Converter;
006    import java.util.Locale;
007    
008    /**
009     * org.doc4web.grails.jsf.faces
010     * <p/>
011     * Date: 23 mars 2010
012     * Time: 12:58:08
013     *
014     * @author Stephane MALDINI
015     */
016    public class LocaleConverter implements Converter {
017        /**
018         * <p>Convert the specified string value, which is associated with
019         * the specified {@link javax.faces.component.UIComponent}, into a model data object that
020         * is appropriate for being stored during the <em>Apply Request
021         * Values</em> phase of the request processing lifecycle.</p>
022         *
023         * @param context   {@link javax.faces.context.FacesContext} for the request being processed
024         * @param component {@link javax.faces.component.UIComponent} with which this model object
025         *                  value is associated
026         * @param value     String value to be converted (may be <code>null</code>)
027         * @return <code>null</code> if the value to convert is <code>null</code>,
028         *         otherwise the result of the conversion
029         * @throws javax.faces.convert.ConverterException
030         *                              if conversion cannot be successfully
031         *                              performed
032         * @throws NullPointerException if <code>context</code> or
033         *                              <code>component</code> is <code>null</code>
034         */
035        public Object getAsObject(FacesContext context, UIComponent component, String value) {
036            String[] values = value.split("_");
037            if (values.length > 2)
038                return new Locale(values[0], values[1]);  //To change body of implemented methods use File | Settings | File Templates.
039            else
040                return new Locale(value);
041        }
042    
043        /**
044         * <p>Convert the specified model object value, which is associated with
045         * the specified {@link javax.faces.component.UIComponent}, into a String that is suitable
046         * for being included in the response generated during the
047         * <em>Render Response</em> phase of the request processing
048         * lifeycle.</p>
049         *
050         * @param context   {@link javax.faces.context.FacesContext} for the request being processed
051         * @param component {@link javax.faces.component.UIComponent} with which this model object
052         *                  value is associated
053         * @param value     Model object value to be converted
054         *                  (may be <code>null</code>)
055         * @return a zero-length String if value is <code>null</code>,
056         *         otherwise the result of the conversion
057         * @throws javax.faces.convert.ConverterException
058         *                              if conversion cannot be successfully
059         *                              performed
060         * @throws NullPointerException if <code>context</code> or
061         *                              <code>component</code> is <code>null</code>
062         */
063        public String getAsString(FacesContext context, UIComponent component, Object value) {
064            return value.toString();
065        }
066    }
