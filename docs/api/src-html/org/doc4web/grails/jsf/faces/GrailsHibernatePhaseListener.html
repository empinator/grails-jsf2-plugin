001    package org.doc4web.grails.jsf.faces;
002    
003    import org.hibernate.FlushMode;
004    import org.hibernate.Session;
005    import org.hibernate.SessionFactory;
006    import org.springframework.orm.hibernate3.SessionFactoryUtils;
007    import org.springframework.orm.hibernate3.SessionHolder;
008    import org.springframework.transaction.support.TransactionSynchronizationManager;
009    
010    import javax.faces.event.PhaseEvent;
011    import javax.faces.event.PhaseId;
012    import javax.faces.event.PhaseListener;
013    
014    /**
015     * org.doc4web.grails.jsf.faces
016     * <p/>
017     * Date: 21 mars 2010
018     * Time: 02:40:30
019     *
020     * @author Stephane MALDINI
021     */
022    public class GrailsHibernatePhaseListener implements PhaseListener {
023    
024    
025        /**
026         * <p>Handle a notification that the processing for a particular
027         * phase has just been completed.</p>
028         */
029    
030    
031        public void afterPhase(PhaseEvent event) {
032            this.endSession(event);
033        }
034    
035        /**
036         * <p>Handle a notification that the processing for a particular
037         * phase of the request processing lifecycle is about to begin.</p>
038         */
039        public void beforePhase(PhaseEvent event) {
040            this.initSession(event);
041        }
042    
043        /**
044         * <p>Return the identifier of the request processing phase during
045         * which this listener is interested in processing {@link javax.faces.event.PhaseEvent}
046         * events.  Legal values are the singleton instances defined by the
047         * {@link javax.faces.event.PhaseId} class, including <code>PhaseId.ANY_PHASE</code>
048         * to indicate an interest in being notified for all standard phases.</p>
049         */
050        public PhaseId getPhaseId() {
051            return PhaseId.ANY_PHASE;
052        }
053    
054        private static ThreadLocal<Boolean> boundByMe = new ThreadLocal<Boolean>() {
055            protected synchronized Boolean initialValue() {
056                return false;
057            }
058        };
059    
060    
061        private void initSession(PhaseEvent event) {
062            if (event.getPhaseId().equals(PhaseId.RESTORE_VIEW))
063                boundByMe.set(bindSession());
064        }
065    
066        private void endSession(PhaseEvent event) {
067            if (event.getPhaseId().equals(PhaseId.RENDER_RESPONSE)) {
068                final SessionHolder sessionHolder = (SessionHolder) TransactionSynchronizationManager.getResource(sessionFactory);
069                if (sessionHolder != null && !FlushMode.MANUAL.equals(sessionHolder.getSession().getFlushMode())) {
070                    sessionHolder.getSession().flush();
071                }
072                if (boundByMe.get()) unbindSession();
073            }
074    
075        }
076    
077        public void setSessionFactory(SessionFactory sessionFactory) {
078            this.sessionFactory = sessionFactory;
079        }
080    
081        private SessionFactory sessionFactory;
082    
083    
084        private boolean bindSession() {
085            if (sessionFactory == null) {
086                throw new IllegalStateException("No sessionFactory property provided");
087            }
088            final Object inStorage = TransactionSynchronizationManager.getResource(sessionFactory);
089            if (inStorage != null) {
090                ((SessionHolder) inStorage).getSession().flush();
091                return false;
092            } else {
093                Session session = SessionFactoryUtils.getSession(sessionFactory, true);
094                session.setFlushMode(FlushMode.AUTO);
095                TransactionSynchronizationManager.bindResource(sessionFactory, new SessionHolder(session));
096                return true;
097            }
098        }
099    
100        private void unbindSession() {
101            if (sessionFactory == null) {
102                throw new IllegalStateException("No sessionFactory property provided");
103            }
104            SessionHolder sessionHolder = (SessionHolder) TransactionSynchronizationManager.getResource(sessionFactory);
105            try {
106                if (!FlushMode.MANUAL.equals(sessionHolder.getSession().getFlushMode())) {
107                    sessionHolder.getSession().flush();
108                }
109            } catch (Exception e) {
110                e.printStackTrace();
111            } finally {
112                TransactionSynchronizationManager.unbindResource(sessionFactory);
113                SessionFactoryUtils.closeSession(sessionHolder.getSession());
114            }
115        }
116    }
