019    public class ViewScope implements Scope {
020    
021            public final String VIEW_SCOPE_KEY = "JSF2_VIEW_SCOPE";

023        private long timestamp = 0;
024

026            return timestamp;

033            public Object get(String name, ObjectFactory objectFactory) {
034    
035                    if (FacesContext.getCurrentInstance().getViewRoot() != null) {
036                            Map<String, Object> viewScope = extractViewScope();
037    
038                            if (viewScope.get(name) == null) {
039                                    Object object = objectFactory.getObject();
040                                    viewScope.put(name, object);
041                                    return object;
042                            } else {
043                                    return viewScope.get(name);
044                            }
045                    } else {
046                            return null;
047                    }
048    
049            }
050    
051            @SuppressWarnings("unchecked")
052            private Map<String, Object> extractViewScope() {
053                    Map<String, Object> attributes = FacesContext.getCurrentInstance().getViewRoot().getAttributes();
054    
055                    Map<String, Object> viewScope = null;
056    
057                    if (attributes.get(VIEW_SCOPE_KEY)==null) {
058                            viewScope = new HashMap<String, Object>();
059                            attributes.put(VIEW_SCOPE_KEY, viewScope);
060                    } else {
061                            viewScope = (Map<String, Object>) attributes.get(VIEW_SCOPE_KEY);
062                    }
063                    return viewScope;
064            }

085                    if (FacesContext.getCurrentInstance().getViewRoot() != null) {
